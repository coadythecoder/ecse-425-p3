        when IDLE:
            if s_read == 1:
                store read_address
                state == READ_INIT
            elif s_write == 1:
                store write_address, write_data
                state == WRITE_INIT
            else:
                do nothing
        when READ_INIT:
            if cache_hit():
                return read_data
            else:
                initialize counter variables for loading bytes from memory (byte_counter = 0)

                if cache[block_index].valid == 1 and cache[block_index].dirty==1:
                    # need to write-back block to memory
                    send first bite to --> m_writedata
                    initialize counter for sending word to memory
                    state == READ_CALLBACK

                state = READ_WAIT
        when WRITE_INIT:
            if cache_hit():
                modify block in place
                set dirty bit to 1
            else
                initialize counter variables for loading bytes from memory (byte_counter = 0)

                if cache[block_index].valid == 1 and cache[block_index].dirty==1:
                    # need to write-back block to memory
                    send first bite to --> m_writedata
                    initialize counter for sending word to memory
                    state == WRITE_CALLBACK

                state = WRITE_WAIT

        when READ_WAIT:
            if byte_counter <= 15:
                if memory_waitrequest == 0:
                    cache[block_index].word[byte_counter/4][(byte_counter%4)*8-1:(byte_counter%4)*8] = memory_readdata
                    byte_counter++
            if byte_counter == 15:
                byte_counter = 0
                state = IDLE
                s_waitrequest = 0
        when WRITE_WAIT:
            if byte_counter <= 3:
                if memory_waitrequest == 0:

            if byte_counter == 3:
                byte_counter = 0
                state = IDLE
                s_waitrequest = 0
